/------ studentinfo table ----/
CREATE TABLE studentinfo(
    title VARCHAR(5),
    fname VARCHAR(25),
    middleName VARCHAR(25),
    lname VARCHAR(25),
    gender VARCHAR(10),
    mobile VARCHAR(12),
    phone VARCHAR(12),
    email VARCHAR(50) UNIQUE,
    dob DATE,
    pob VARCHAR(20),
    address VARCHAR(100),
    marital_status VARCHAR(20),
    fatherName VARCHAR(50),
    fOccupation VARCHAR(30),
    motherName VARCHAR(50),
    mOccupation VARCHAR(30),
    parentPhone VARCHAR(12),
    parentAddress VARCHAR(100),
    casteCategory VARCHAR(30),
    sub_caste VARCHAR(30),
    nationality VARCHAR(50),
    religion VARCHAR(30),
    handicap VARCHAR(10),
    up_aadhar VARCHAR(50) UNIQUE,
    up_pan VARCHAR(50) UNIQUE,
    up_tenth VARCHAR(50) UNIQUE,
    up_twelveth VARCHAR(50) UNIQUE,
    PRN VARCHAR(15) UNIQUE,
    COURSE VARCHAR(15),
    STD_YEAR VARCHAR(20),
    DIVISION VARCHAR(10),
    ROLLNO VARCHAR(10)
    );
